Embedding a Whole File

To embed an entire file into a FubuMVC page, just use the following extension methods on IFubuPage: 
  1.         /// <summary>
  2.         /// Embed the contents of an entire code file into a FubuMVC view
  3.         /// </summary>
  4.         /// <param name="page"></param>
  5.         /// <param name="fileName">The name of the file relative to the root of the web application or Bottle</param>
  6.         /// <param name="languageClass">Optional.  Overrides the lang-* class on the element for prettify.css formatting</param>
  7.         /// <param name="levelIndentation">Not yet used.  Will force the display to remove extra indentation for deeply nested code</param>
  8.         /// <returns></returns>
  9.         public static HtmlTag CodeFile(this IFubuPage page, string fileName, string languageClass = null, bool levelIndentation = false)
  10.         {
  11.             var formatter = new CodeFormatter(levelIndentation);
  12.             var file = page.Get<IFubuApplicationFiles>().Find(fileName);
  13.  
  14.             if (file == null)
  15.             {
  16.                 return new HtmlTag("p").Text("Could not find file " + fileName);
  17.             }
  18.  
  19.             var snippet = formatter.Format(file);
  20.  
  21.             return page.CodeSnippet(snippet);
  22.         }
Snippet SAMPLE/ENDSAMPLE comments are stripped out of the rendered content and replaced with a blank line to keep the line numbers consistent with the file.

Linking to a File

You can also provide a link to a code file that will be rendered with prettyprint in a new window by using one of these two extension methods: 
  1.         /// <summary>
  2.         /// Writes a link to a page that will display the entire contents of the file 
  3.         /// containing the named snippet
  4.         /// </summary>
  5.         /// <param name="page"></param>
  6.         /// <param name="snippetName"></param>
  7.         /// <returns></returns>
  8.         public static HtmlTag LinkToSnippet(this IFubuPage page, string snippetName)
  9.         {
  10.             var snippet = page.Get<ISnippetCache>().Find(snippetName);
  11.             var fileName = snippet.File;
  12.  
  13.             return page.LinkToCodeFile(fileName);
  14.         }
  15.  
  16.         /// <summary>
  17.         /// Writes a link to a page that will display the entire contents of a code file
  18.         /// </summary>
  19.         /// <param name="page"></param>
  20.         /// <param name="fileName">The name of the file relative to the root of the web application or Bottle</param>
  21.         /// <returns></returns>
  22.         public static HtmlTag LinkToCodeFile(this IFubuPage page, string fileName)
  23.         {
  24.             var url = page.Urls.UrlFor(new CodeFileRequest(fileName));
  25.             return new LinkTag(Path.GetFileName(fileName), url, "code-link").Title(fileName).Attr("target", "_blank");
  26.         }
